---
title: Modal
---

A modal is an overlay element which blocks interaction with elements outside it.

<Example>
  <DialogTrigger>
    <Button>Center</Button>
    <Modal>
      <Dialog>
        <Flexbox justifyContent="center" alignItems="center" paddingX="9" paddingY="8">
          <Heading>Hello.</Heading>
        </Flexbox>
      </Dialog>
    </Modal>
  </DialogTrigger>
  <DialogTrigger>
    <Button>Top</Button>
    <Modal position="top">
      <Dialog variant="sheet">
        <Flexbox justifyContent="center" alignItems="center" paddingX="9" paddingY="8">
          <Heading>Hello.</Heading>
        </Flexbox>
      </Dialog>
    </Modal>
  </DialogTrigger>
  <DialogTrigger>
    <Button>Left</Button>
    <Modal position="left">
      <Dialog variant="sheet">
        <Flexbox justifyContent="center" alignItems="center" paddingX="9" paddingY="8">
          <Heading>Hello.</Heading>
        </Flexbox>
      </Dialog>
    </Modal>
  </DialogTrigger>
  <DialogTrigger>
    <Button>Right</Button>
    <Modal position="right">
      <Dialog variant="sheet">
        <Flexbox justifyContent="center" alignItems="center" paddingX="9" paddingY="8">
          <Heading>Hello.</Heading>
        </Flexbox>
      </Dialog>
    </Modal>
  </DialogTrigger>
  <DialogTrigger>
    <Button>Bottom</Button>
    <Modal position="bottom">
      <Dialog variant="sheet">
        <Flexbox justifyContent="center" alignItems="center" paddingX="9" paddingY="8">
          <Heading>Hello.</Heading>
        </Flexbox>
      </Dialog>
    </Modal>
  </DialogTrigger>
  <DialogTrigger>
    <Button>Fullscreen</Button>
    <Modal position="fullscreen">
      <Dialog variant="sheet">
        <Flexbox justifyContent="center" alignItems="center" paddingX="9" paddingY="8">
          <Heading>Hello.</Heading>
        </Flexbox>
      </Dialog>
    </Modal>
  </DialogTrigger>
</Example>
